class Config(object):
    """
    store information in a faster rcnn model
    Idea from keras-frcnn(https://github.com/jinfagang/keras_frcnn)
    """
    def __init__(self, raw_feature_extraction_net='vgg',v=True):
        # initialize raw feature extraction net
        self.network = raw_feature_extraction_net
        self.verbose = v
        # setting for data augmentation
        self.use_horizontal_flips = False
        self.use_vertical_flips = False
        self.rot_90 = False

        # anchor box scales
        self.anchor_box_scales = [128, 256, 512]

        # anchor box ratios
        self.anchor_box_ratios = [[1, 1], [1./np.sqrt(2), 2./np.sqrt(2)], [2./np.sqrt(2), 1./np.sqrt(2)]]

        # size to resize the smallest side of the image
        self.im_size = 600

        # image channel-wise mean to subtract
        self.img_channel_mean = [103.939, 116.779, 123.68]
        self.img_scaling_factor = 1.0

        # number of ROIs at once
        self.num_rois = 4

        # stride at the RPN 
        self.rpn_stride = 16
        self.balanced_classes = False

        # scaling the stdev
        self.std_scaling = 4.0
        self.classifier_regr_std = [8.0, 8.0, 4.0, 4.0]

        # overlaps for RPN
        self.rpn_min_overlap = 0.3
        self.rpn_max_overlap = 0.7

        # overlaps for classifier ROIs
        self.classifier_min_overlap = 0.1
        self.classifier_max_overlap = 0.5

        # placeholder for the class mapping, automatically generated by the parser
        self.class_mapping = None
        self.model_path = 'model_frcnn.vgg.hdf5'

def RPN(base_layers, num_anchors):
    """
    inputs:
        base_layers: feature map, output from shared CNN (raw feature extraction net)
        num_anchors: in faster rcnn, equals to 3 x 3
    output: 
        x_cls: 9-d, used to build cls_loss, cross entropy
        x_regr: 4x9-d, used to build regr_loss, smooth L1
    """
    x = Conv2D(512, (3, 3), padding='same', activation='relu', kernel_initializer='normal', name='rpn_conv1')(base_layers)
    x_cls = Conv2D(num_anchors, (1, 1), activation='sigmoid', kernel_initializer='uniform', name='rpn_out_class')(x)
    x_regr = Conv2D(num_anchors * 4, (1, 1), activation='linear', kernel_initializer='zero', name='rpn_out_regress')(x)
    return [x_cls, x_regr, base_layers]

def classifier_vgg(base_layers, input_rois, num_rois, nb_classes = 21, trainable=False):
    """
    input:
        base_layers: output from shared CNN.
        input_rois: impact on the feature map
        nb_classes: 20 + 1(bp) = 21
    """
    # we try theano and tensorflow, tensorflow is much better
    if K.backend() == 'tensorflow':
        pooling_regions = 7
        input_shape = (num_rois,7,7,512)
    elif K.backend() == 'theano':
        pooling_regions = 7
        input_shape = (num_rois,512,7,7)

    # Roipooling layer
    out_roi_pool = RoiPoolingConv(pooling_regions, num_rois)([base_layers, input_rois])
    # fc1 + relu1 layer
    out = TimeDistributed(Flatten(name='flatten'))(out_roi_pool)
    out = TimeDistributed(Dense(4096, activation='relu', name='fc1'))(out)
    # dropout1 layer
    out = TimeDistributed(Dropout(0.5))(out)
    # fc2 + relu2 layer
    out = TimeDistributed(Dense(4096, activation='relu', name='fc2'))(out)
    # dropout2 layer
    out = TimeDistributed(Dropout(0.5))(out)
    
    # output layer
    out_cls = TimeDistributed(Dense(nb_classes, activation='softmax', kernel_initializer='zero'), name='dense_class_{}'.format(nb_classes))(out)
    # note: no regression target for bg class
    out_regr = TimeDistributed(Dense(4 * (nb_classes-1), activation='linear', kernel_initializer='zero'), name='dense_regress_{}'.format(nb_classes))(out)

    return [out_cls, out_regr]

def build_model(C,image_input,roi_input,classes_count,nn):
    """
    C: config defined before
    image_input: image data into shared CNN
    roi_input: roi input, impact on the feature map
    nn are function name
    """
    # First part: shared CNN, could be vgg16, resnet50, inception
    shared_CNN = nn.nn_base(image_input,trainable=True)
    # add RPN part
    num_anchors = len(C.anchor_box_scales) * len(C.anchor_box_ratios)
    rpn = RPN(shared_CNN, num_anchors)
    # add classifier (fast RCNN) part
    if C.network == 'vgg':
        classifier = classifier_vgg(shared_CNN, roi_input,
                     C.num_rois, nb_classes=len(classes_count),trainable=True)
    else:
        # resnet or inception
        classifier = nn.classifier(shared_CNN, roi_input, C.num_rois, nb_classes=len(classes_count), trainable=True)
    model_rpn = Model(image_input,rpn[:2])
    model_classifier = Model([image_input, roi_input], classifier)
    # combine two model
    model_all = Model([image_input, roi_input], rpn[:2] + classifier)
    return model_rpn, model_classifier, model_all

        pickle.dump(C,config_f)
        print('Config has been written to {}, and can be loaded when testing to ensure correct results'.format(config_output_filename))


    # --------------------------------------------------------
    # Phase two: train the model
    # --------------------------------------------------------
    from keras_frcnn import losses as losses

    # shuffle input data
    random.shuffle(all_imgs)    
    num_imgs = len(all_imgs)

    # train image and validation image
    train_imgs = [s for s in all_imgs if s['imageset'] == 'trainval']
    val_imgs = [s for s in all_imgs if s['imageset'] == 'test']
    print('Number of training samples {}'.format(len(train_imgs)))
    print('Number of val samples {}'.format(len(val_imgs)))

    model_all.compile(optimizer='sgd', loss='mae')

    #### epoches
    epoch_length = 1000
    num_epochs = int(options.num_epochs)
    iter_num = 0
    #### use numpy to store losses
    losses = np.zeros((epoch_length, 5))

    rpn_accuracy_for_epoch = []
    start_time = time.time()

    best_loss = np.Inf # threshold

    class_mapping_inv = {v: k for k, v in class_mapping.items()}
    print('Starting training')

    vis = Tru
    for epoch_num in range(num_epochs):

        progbar = generic_utils.Progbar(epoch_length)
        print('Epoch {}/{}'.format(epoch_num + 1, num_epochs))

        while True:
            """
            IDEA used here
            Reference:  https://github.com/riadhayachi/faster-rcnn-keras
            """

            if len(rpn_accuracy_rpn_monitor) == epoch_length and C.verbose:
                mean_overlapping_bboxes = float(sum(rpn_accuracy_rpn_monitor))/len(rpn_accuracy_rpn_monitor)
                rpn_accuracy_rpn_monitor = []
                print('Average number of overlapping bounding boxes from RPN = {} for {} previous iterations'.format(mean_overlapping_bboxes, epoch_length))
                if mean_overlapping_bboxes == 0:
                    print('RPN is not producing bounding boxes that overlap the ground truth boxes. Check RPN settings or keep training.')

            ##### generator get new data #####
            X, Y, img_data = next(data_gen_train)
            # RPN loss
            loss_rpn = model_rpn.train_on_batch(X, Y)
            # RPN proposal
            P_rpn = model_rpn.predict_on_batch(X)
            # format transition: calc_iou converts from (x1,y1,x2,y2) to (x,y,w,h) format
            R = roi_helpers.rpn_to_roi(P_rpn[0], P_rpn[1], C, K.image_dim_ordering(), use_regr=True, overlap_thresh=0.7, max_boxes=300)
            X2, Y1, Y2, IouS = roi_helpers.calc_iou(R, img_data, C, class_mapping)

            if X2 is None:
                rpn_accuracy_rpn_monitor.append(0)
                rpn_accuracy_for_epoch.append(0)
                continue

            ##### judge1 #####
            neg_samples = np.where(Y1[0, :, -1] == 1)
            pos_samples = np.where(Y1[0, :, -1] == 0)

            ##### judge2 #####
            if len(neg_samples) > 0:
                neg_samples = neg_samples[0]
            else:
                neg_samples = []

            if len(pos_samples) > 0:
                pos_samples = pos_samples[0]
            else:
                pos_samples = []

            # rpn accuracy
            rpn_accuracy_rpn_monitor.append(len(pos_samples))
            rpn_accuracy_for_epoch.append((len(pos_samples)))

            ##### make sure rois number is larger than 1 #####
            if C.num_rois > 1:
                if len(pos_samples) < C.num_rois//2:
                    selected_pos_samples = pos_samples.tolist()
                else:
                    selected_pos_samples = np.random.choice(pos_samples, C.num_rois//2, replace=False).tolist()
                try:
                    selected_neg_samples = np.random.choice(neg_samples, C.num_rois - len(selected_pos_samples), replace=False).tolist()
                except:
                    selected_neg_samples = np.random.choice(neg_samples, C.num_rois - len(selected_pos_samples), replace=True).tolist()

                sel_samples = selected_pos_samples + selected_neg_samples
            else:
                selected_pos_samples = pos_samples.tolist()
                selected_neg_samples = neg_samples.tolist()
                if np.random.randint(0, 2):
                    sel_samples = random.choice(neg_samples)
                else:
                    sel_samples = random.choice(pos_samples)

            ##### approximating joint training #####
            loss_class = model_classifier.train_on_batch([X, X2[:, sel_samples, :]], [Y1[:, sel_samples, :], Y2[:, sel_samples, :]])

            losses[iter_num, 0] = loss_rpn[1]
            losses[iter_num, 1] = loss_rpn[2]
            losses[iter_num, 2] = loss_class[1]
            losses[iter_num, 3] = loss_class[2]
            losses[iter_num, 4] = loss_class[3]

            ##### updating #####
            iter_num += 1

            progbar.update(iter_num, [('rpn_cls', np.mean(losses[:iter_num, 0])), ('rpn_regr', np.mean(losses[:iter_num, 1])),
                                      ('detector_cls', np.mean(losses[:iter_num, 2])), ('detector_regr', np.mean(losses[:iter_num, 3]))])


            ##### stopping condition ##### 
            if iter_num == epoch_length:
                loss_rpn_cls = np.mean(losses[:, 0])
                loss_rpn_regr = np.mean(losses[:, 1])
                loss_class_cls = np.mean(losses[:, 2])
                loss_class_regr = np.mean(losses[:, 3])
                class_acc = np.mean(losses[:, 4])

                mean_overlapping_bboxes = float(sum(rpn_accuracy_for_epoch)) / len(rpn_accuracy_for_epoch)
                rpn_accuracy_for_epoch = []

                if C.verbose:
                    print('Mean number of bounding boxes from RPN overlapping ground truth boxes: {}'.format(mean_overlapping_bboxes))
                    print('Classifier accuracy for bounding boxes from RPN: {}'.format(class_acc))
                    print('Loss RPN classifier: {}'.format(loss_rpn_cls))
                    print('Loss RPN regression: {}'.format(loss_rpn_regr))
                    print('Loss Detector classifier: {}'.format(loss_class_cls))
                    print('Loss Detector regression: {}'.format(loss_class_regr))
                    print('Elapsed time: {}'.format(time.time() - start_time))

                curr_loss = loss_rpn_cls + loss_rpn_regr + loss_class_cls + loss_class_regr
                iter_num = 0
                start_time = time.time()

                if curr_loss < best_loss:
                    if C.verbose:
                        print('Total loss decreased from {} to {}, saving weights'.format(best_loss,curr_loss))
                    best_loss = curr_loss
                    model_all.save_weights(C.model_path)

                break
    print('Training complete, exiting.')
